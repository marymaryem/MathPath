def ondulant (x):
    ch=str(x)
    valid=True
    for i in range(len(ch)-2):
         if not(ch[i]==ch[i+2] and ch[i]!=ch[i+1]):
              valid=False
    return valid

def premier(x):
    ok=True
    for i in range(2,x//2):
       if x%i==0:
       ok=False
    return ok

def lisse(x):
    pg=2
    for i in range(2,x):
        if x%i==0:
            if premier(i) and i > pg:
                 pg = i
    return pg <= sqrt(x)

def semi_premier(x):
    v=False
    for i in range (2,x):
       if premier(i) and premier(x // 2) and i * (x // i) == x:
        v=True
    return v

def parfait(x):
    s=0
    for i in range(1,(x//2) + 1):
         if x%i==0: s=s+i
    return s==x

def pgcd_difference(x,y):
    while (x != y) :
        if (x < y): y = y - x
        else : x = x - y
    return x


def pgcd_euclide(x,y):
    while (x % y != 0) :
        r = x % y
        x = y
        y = r
    return y

def ppcm(x,y):
    xa = x
    while (x % y != 0) :
         x = x + xa
    return x

def convert_from_base(ch, b):
    n = 0
    p = 1
    for i in range(len(ch)-1, -1, -1):
         x = ch[i]
         if "0" <= x <= "9":
              n = n + int(x) * p
         else:
              n = n + (ord(x) - 55) * p
         p *= b
    return n



def convert_to_base(n, b):
    ch = ""
    while n != 0:
        r = n % b
        n = n // b
        if r < 10:
              ch = str(r) + ch
         else:
              ch = chr(r + 55) + ch
    return ch
